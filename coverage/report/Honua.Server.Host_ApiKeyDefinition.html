 207    /// through timing side-channels. An attacker could measure response times to determine
 208    /// which characters in their guess are correct, gradually revealing the full key.
 209    ///

 211    /// preventing this attack vector.

 213    private static bool IsApiKeyMatch(string configuredKey, string providedKey)
 214    {

 218        // Convert both keys to byte arrays for constant-time comparison

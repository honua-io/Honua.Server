 933        private static NodeItem[] BuildTree(List<NodeItem> leafNodes, NodeItem extent, ushort nodeSize)
 934        {
 935            var branchingFactor = Math.Clamp(nodeSize, (ushort)2, (ushort)65535);
 936            var levelBounds = GenerateLevelBounds(leafNodes.Count, branchingFactor);
 937            var numNodes = levelBounds[0].end;
 938            var nodeItems = new NodeItem[numNodes];
 939            var leafStart = numNodes - leafNodes.Count;
 940
 941            for (var i = 0; i < leafNodes.Count; i++)
 942            {
 943                nodeItems[leafStart + i] = leafNodes[i];
 944            }
 945
 946            for (var level = 0; level < levelBounds.Count - 1; level++)
 947            {
 948                var childrenRange = levelBounds[level];
 949                var parentRange = levelBounds[level + 1];
 950                var parentIndex = parentRange.start;
 951                var childIndex = childrenRange.start;
 952
 953                while (childIndex < childrenRange.end)
 954                {
 955                    var parentNode = NodeItem.CreateInternal((ulong)childIndex);
 956                    for (int j = 0; j < branchingFactor && childIndex < childrenRange.end; j++, childIndex++)
 957                    {
 958                        parentNode.Expand(nodeItems[childIndex]);
 959                    }
 960
 961                    nodeItems[parentIndex++] = parentNode;
 962                }
 963            }
 964
 965            return nodeItems;
 966        }

 968        private static List<(int start, int end)> GenerateLevelBounds(int numItems, ushort nodeSize)
 969        {
 970            if (numItems <= 0)
 971            {
 972                throw new ArgumentOutOfRangeException(nameof(numItems));
 973            }
 974
 975            var branching = Math.Clamp(nodeSize, (ushort)2, (ushort)65535);
 976            var levelNumNodes = new List<int> { numItems };
 977            var n = numItems;
 978            var totalNodes = n;
 979
 980            while (true)
 981            {
 982                n = (n + branching - 1) / branching;
 983                totalNodes += n;
 984                levelNumNodes.Add(n);
 985                if (n == 1)
 986                {
 987                    break;
 988                }
 989            }
 990
 991            var levelBounds = new List<(int start, int end)>(levelNumNodes.Count);
 992            var current = totalNodes;
 993            foreach (var size in levelNumNodes)
 994            {
 995                current -= size;
 996                levelBounds.Add((current, current + size));
 997            }
 998
 999            return levelBounds;
 1000        }

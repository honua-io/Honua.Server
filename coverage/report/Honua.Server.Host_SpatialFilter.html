 425    private static string? ParseAcceptCrsHeader(StringValues headerValues)
 426    {
 427        if (headerValues.Count == 0)
 428        {
 429            return null;
 430        }
 431
 432        var bestValue = (Value: (string?)null, Weight: double.NegativeInfinity, Order: int.MaxValue);
 433        var order = 0;
 434
 435        foreach (var header in headerValues)
 436        {
 437            if (header.IsNullOrWhiteSpace())
 438            {
 439                continue;
 440            }
 441
 442            var entries = QueryParsingHelpers.ParseCsv(header);
 443            foreach (var entry in entries)
 444            {
 445
 446                var segments = entry.Split(';', StringSplitOptions.RemoveEmptyEntries | StringSplitOptions.TrimEntries);
 447                if (segments.Length == 0)
 448                {
 449                    order++;
 450                    continue;
 451                }
 452
 453                var value = segments[0];
 454                if (!value.IsNullOrEmpty() && value.Length > 1 && value[0] == '"' && value[^1] == '"')
 455                {
 456                    value = value[1..^1];
 457                }
 458
 459                double weight = 1.0;
 460                for (var i = 1; i < segments.Length; i++)
 461                {
 462                    var part = segments[i];
 463                    if (part.StartsWithIgnoreCase("q=") &&
 464                        part.AsSpan(2).TryParseDouble(out var parsedWeight))
 465                    {
 466                        weight = Math.Clamp(parsedWeight, 0d, 1d);
 467                    }
 468                }
 469
 470                if (weight <= 0d)
 471                {
 472                    order++;
 473                    continue;
 474                }
 475
 476                if (string.Equals(value, "*", StringComparison.Ordinal))
 477                {
 478                    order++;
 479                    continue;
 480                }
 481
 482                if (weight > bestValue.Weight || (Math.Abs(weight - bestValue.Weight) <= 1e-6 && order < bestValue.Order
 483                {
 484                    bestValue = (value, weight, order);
 485                }
 486
 487                order++;
 488            }
 489        }
 490
 491        return bestValue.Value;
 492    }

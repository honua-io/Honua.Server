 173        MetadataSnapshot snapshot,
 174        CancellationToken cancellationToken)
 175    {

 177        {
 178            GeneratedAt = DateTime.UtcNow
 179        };

 182        var layersByDataSource = snapshot.Layers
 183            .Join(snapshot.Services,

 185                service => service.Id,

 191            var dataSource = snapshot.DataSources.FirstOrDefault(ds => ds.Id == group.Key);
 192            if (dataSource == null)
 193            {
 194                this.logger.LogWarning("Data source '{DataSourceId}' not found", group.Key);
 195                continue;
 196            }
 197
 198            try
 199            {
 200                var layers = group.Select(x => x.Layer).ToList();
 201                var results = await DiagnoseDataSourceLayersAsync(dataSource, layers, cancellationToken);
 202                report.Results.AddRange(results);
 203            }

 207            }
 208        }
 209
 210        // Calculate summary
 211        report.TotalLayers = report.Results.Count;
 212        report.LayersWithIndexes = report.Results.Count(r => r.HasSpatialIndex);
 213        report.LayersWithoutIndexes = report.TotalLayers - report.LayersWithIndexes;
 214        report.LayersWithIssues = report.Results.Count(r => r.Issues.Count > 0);
 215
 216        return report;

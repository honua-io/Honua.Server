 321    /// Extracts address string from a feature using the configuration.
 322    /// </summary>
 323    private string? ExtractAddressFromFeature(ParsedFeature feature, AddressConfiguration config)
 324    {
 325        if (config.Type == AddressConfigurationType.SingleColumn)
 326        {
 327            if (string.IsNullOrEmpty(config.SingleColumnName))
 328                return null;
 329
 330            return feature.Properties.TryGetValue(config.SingleColumnName, out var value)
 331                ? value?.ToString()
 332                : null;
 333        }
 334        else if (config.Type == AddressConfigurationType.MultiColumn)
 335        {
 336            if (config.MultiColumnNames == null || !config.MultiColumnNames.Any())
 337                return null;
 338
 339            var parts = new List<string>();
 340            foreach (var columnName in config.MultiColumnNames)
 341            {
 342                if (feature.Properties.TryGetValue(columnName, out var value) &&
 343                    !string.IsNullOrWhiteSpace(value?.ToString()))
 344                {
 345                    parts.Add(value.ToString()!);
 346                }
 347            }
 348
 349            return parts.Any() ? string.Join(config.Separator, parts) : null;
 350        }
 351
 352        return null;

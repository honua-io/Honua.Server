 202    private static string SanitizeIdentifier(string input)
 203    {
 204        // Remove invalid characters and convert to lowercase
 205        var sanitized = new StringBuilder();
 206        foreach (var c in input)
 207        {
 208            if (char.IsLetterOrDigit(c) || c == '_')
 209            {
 210                sanitized.Append(char.ToLowerInvariant(c));
 211            }
 212            else if (c == ' ' || c == '-')
 213            {
 214                sanitized.Append('_');
 215            }
 216        }
 217
 218        var result = sanitized.ToString();
 219
 220        // Ensure it doesn't start with a digit
 221        if (result.Length > 0 && char.IsDigit(result[0]))
 222        {
 223            result = "_" + result;
 224        }
 225
 226        return result;
 227    }

 229    private static string ToTitleCase(string input)
 230    {
 231        // Convert snake_case or PascalCase to Title Case
 232        var words = input.Split('_', ' ', '-')
 233            .Where(w => !string.IsNullOrWhiteSpace(w))
 234            .Select(w => char.ToUpperInvariant(w[0]) + w.Substring(1).ToLowerInvariant());

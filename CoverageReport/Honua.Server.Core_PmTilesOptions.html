 305    private static byte[] CompressData(byte[] data, PmTilesCompression compression)
 306    {
 307        if (compression == PmTilesCompression.None || data.Length == 0)
 308        {
 309            return data;
 310        }
 311
 312        using var inputStream = new MemoryStream(data);
 313        using var outputStream = new MemoryStream();
 314
 315        switch (compression)
 316        {
 317            case PmTilesCompression.Gzip:
 318                using (var gzipStream = new GZipStream(outputStream, CompressionLevel.Optimal, leaveOpen: true))
 319                {
 320                    inputStream.CopyTo(gzipStream);
 321                }
 322                break;
 323
 324            case PmTilesCompression.Brotli:
 325                using (var brotliStream = new BrotliStream(outputStream, CompressionLevel.Optimal, leaveOpen: true))
 326                {
 327                    inputStream.CopyTo(brotliStream);
 328                }
 329                break;
 330
 331            case PmTilesCompression.Zstd:
 332                using (var zstdStream = new CompressionStream(outputStream, leaveOpen: true))
 333                {
 334                    inputStream.CopyTo(zstdStream);
 335                }
 336                break;
 337
 338            default:
 339                return data;
 340        }
 341
 342        return outputStream.ToArray();

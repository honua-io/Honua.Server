================================================================================
  HONUA GIS SERVER - SEEDED DOCKER COMPOSE QUICKSTART
================================================================================

FASTEST START (3 commands):
  cd /home/mike/projects/HonuaIO
  docker-compose -f docker-compose.seed.yml up
  # Wait for "Database seeding completed successfully!"

WHAT YOU GET:
  ✓ PostgreSQL 16 + PostGIS 3.4
  ✓ Honua GIS Server (all APIs enabled)
  ✓ Sample GIS data (features, geometries, metadata)
  ✓ All OGC standards (WFS, WMS, WCS, OGC API)
  ✓ STAC 1.0 catalog
  ✓ GeoServices REST API

ACCESS THE SERVER:
  http://localhost:8080

SAMPLE ENDPOINTS:
  Health:       http://localhost:8080/health
  OGC API:      http://localhost:8080/
  Collections:  http://localhost:8080/collections
  WFS:          http://localhost:8080/wfs?service=WFS&request=GetCapabilities
  WMS:          http://localhost:8080/wms?service=WMS&request=GetCapabilities
  STAC:         http://localhost:8080/stac
  GeoServices:  http://localhost:8080/rest/services

QUICK TESTS:
  # Check health
  curl http://localhost:8080/health

  # List collections
  curl http://localhost:8080/collections | jq .

  # Get features
  curl "http://localhost:8080/collections/test_features/items?limit=5" | jq .

  # Run automated tests
  ./scripts/test-seeded-deployment.sh

DATABASE ACCESS:
  docker exec -it honua-postgres-seed psql -U honua -d honua

MANAGEMENT:
  # Start in background
  docker-compose -f docker-compose.seed.yml up -d

  # View logs
  docker-compose -f docker-compose.seed.yml logs -f

  # Stop (keep data)
  docker-compose -f docker-compose.seed.yml down

  # Stop and remove all data
  docker-compose -f docker-compose.seed.yml down -v

  # Restart
  docker-compose -f docker-compose.seed.yml restart

QGIS CONNECTION:
  WFS: http://localhost:8080/wfs
  WMS: http://localhost:8080/wms

CONFIGURATION:
  Edit .env.seed to customize:
    - Database credentials
    - Server port
    - Authentication mode
    - Logging levels
    - Performance settings

TROUBLESHOOTING:
  # View logs
  docker-compose -f docker-compose.seed.yml logs

  # Check service status
  docker-compose -f docker-compose.seed.yml ps

  # Fresh start (removes all data!)
  docker-compose -f docker-compose.seed.yml down -v
  docker-compose -f docker-compose.seed.yml up
